# 通过十道Promise练习题，看Promise特点

# 题目一

const promise = new Promise((resolve, reject) => {
  console.log(1)
  resolve()
  console.log(2)
})
promise.then(() => {
  console.log(3)
})
console.log(4)

这题考察的主要是Promise内部的执行顺序，在之前我们通过源码了解Promsie的时候我们就了解到，我们传入Promise构造函数的是一个执行器(executor)，这个执行器是同步调用的，而Promise的then方法需要等待Promise状态的改变才会执行里面的回调函数，而且回调函数外部包了一层setTimeout。

function Promise(executor){
  ...
  function resolve(value){
    if(this.status == 'pending'){
      this.status = 'fulfilled'
      this.value = value
    }
  }
  ...
  try{
    executor(resolve,reject)
  }catch(err){
      reject(err)
  }
}
Promise.prototype.then = function(onFulfilled,onRejected){
  let self = this
  ...
    return new Promise(resolve,reject){
    if(self.status = 'fulfilled')
      setTimeout(()=> {
          onFulfilled(self.value)
      })
    ...
  }
}
//所以答案当然就是
 1
 2
 4
 3

# 题目二

const promise = new Promise((resolve, reject) => {
  resolve('success1')
  reject('error')
  resolve('success2')
})

promise
  .then((res) => {
    console.log('then: ', res)
  })
  .catch((err) => {
    console.log('catch: ', err)
  })

Promise的状态，只能由pending -> fulfilled 或者由 pending -> rejected，状态一旦改变就不能变更。

function Promise(executor){
  this.status = 'pending'
  ...
  function resolve(value){
    if(this.status == 'pending'){
      this.status =  'fulfilled'
    }
  function reject(reason){
      if(this.status == 'pending'){
          this.status = 'rejected'
      }
  }
  }
// then: success1

# 题目三

Promise.resolve(1)
  .then((res) => {
    console.log(res)
    return 2
  })
  .catch((err) => {
    return 3
  })

    console.log(res)
  })

在Promise的then方法中每次都返回了一个新的Promise，如果函数执行成功，那么就把回调函数的结果传给新Promise的resolve函数

Promise.prototype.then = function () {
    return new Promise((reslove, reject){
        if (self.status == 'fulfilled') {
            setTimeout(() => {
                try {

                    resolvePromise(Promise2, x, resolve, reject)

}

reslovePromise(Promise2,x,resolve,reject){
    ...

# 题目四

Promise.resolve()
  .then(() => {
    return new Error('error!!!')
  })
  .then((res) => {
    console.log('then: ', res)
  })
  .catch((err) => {
    console.log('catch: ', err)
  })

如果在then方法的回调函数中以return的形式返回一个错误对象，并不能被后续的catch方法捕获。

Promise.prototype.then = function () {
  ...
    return new Promise((reslove, reject){
        if (self.status == 'fulfilled') {
            setTimeout(() => {
                try {
                    let x = onFulfilled(self.value) //如果在这里返回的x是一个错误对象，try,catch结构不能捕获，并把错误对象当做正常的结果返回。
                    resolvePromise(Promise2, x, resolve, reject)
                } catch (err) {
                    reject(err)
                }
            })
        }
    })
}
reslovePromise(Promise2,x,resolve,reject){
    ...
    resolve(x) //这样在Promise2中就误把x当做正常参数传入resolve函数了。从而导致catch不能捕获。
}
// then: Error: error!!!
    at Promise.resolve.then (...)
    at ...

如果我们想要Promise能够正常地捕获错误，我们需要改写成这样

return Promise.reject(new Error('error!!!'))
throw new Error('error!!!')

# 题目七

Promise.resolve()
  .then(function success (res) {
    throw new Error('error')
  }, function fail1 (e) {
    console.error('fail1: ', e)
  })
  .catch(function fail2 (e) {
    console.error('fail2: ', e)
  })

Promise.then方法可以接受两个参数，第一个是成功的回调，第二个是错误的回调函数。但在同一个than中第二个处理错误的回调函数是没法捕获第一个成功回调里面抛出的错误的。

这是因为想要调用错误回调，就必须让Promise的状态转变为rejected，then方法当然没有办法让状态已经为fulfilled的Promise再转变状态。

catch方法是一个特殊的than方法，这一点我们已经知道。在then方法中返回一个新的Promise2，那么这时候如果在Promise1的then方法中有抛错，我们就可以让这个新的Promise2的状态改变为rejected。于是catch方法就可以捕获到了。
